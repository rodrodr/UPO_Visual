library(ggthemes)
library(ggplot2)
library(ggiraph)

nc <- sf::st_read(system.file("shape/nc.shp", package = "sf"), quiet = TRUE)

gg <- ggplot(nc) +
  geom_sf_interactive(aes(fill = AREA, tooltip = NAME, data_id = NAME))+
  theme_map()+
  theme(legend.position = "none")+
  scale_fill_continuous_tableau(palette="Purple")
  
  
x <- girafe(ggobj = gg)
